function rotateImages() { const containers = document.querySelectorAll(".image-container"); containers.forEach((container) => { const images = container.querySelectorAll(".rotating-image"); let currentIndex = 0; setInterval(() => { images[currentIndex].classList.remove("active"); currentIndex = (currentIndex + 1) % images.length; images[currentIndex].classList.add("active"); }, 3000); }); } document.addEventListener("DOMContentLoaded", rotateImages);
